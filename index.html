      function displayRecipe() {
        const recipeNumber = window.location.search;
        const urlParams = new URLSearchParams(recipeNumber);
        const recipeURLParams = urlParams.get('number-recipe');

        $.getJSON("recipes.json", function(data) {
          let dataRecipes = data;
          let recipeOutput;
          recipeOutput = "<h3>Ингредиенты:</h3><ul>";
          for (let j in dataRecipes[recipeURLParams].ingredients) {
            recipeOutput += "<li>" + dataRecipes[recipeURLParams].ingredients[j].ingredientName + 
            " - " + dataRecipes[recipeURLParams].ingredients[j].ingredientQuantity
          };
          recipeOutput += "</ul><h3>Рецепт:</h3>";
          recipeOutput += "<p>" + dataRecipes[recipeURLParams].recipe + "</p>";
          document.getElementById('recipe-page').innerHTML=recipeOutput;
        });
      };
      window.onload = displayRecipe;
